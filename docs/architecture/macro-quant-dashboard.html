4. 精度保障 — mathjs 高精度计算

❌ 浮点直接运算 (禁用)

// 浮点误差示例
0.1 + 0.2 === 0.3  // false! (0.30000000000000004)

// GARCH 似然函数中的累积误差
let sigma2 = omega;
for (let t = 1; t < N; t++) {
  // 浮点误差在 10⁵ 次迭代中放大约 10⁻¹²
  sigma2 = omega + alpha * eps2 + beta * sigma2;
}
// 结果偏差: ~1e-8 → 年化波动率偏差 ~0.01%

✅ mathjs BigNumber (推荐)

import { create, all, BigNumber } from 'mathjs';
const math = create(all, { number: 'BigNumber', precision: 64 });

// 配置 64 位精度 → 误差 < 10⁻¹⁹
math.config({ epsilon: 1e-30 });

// GARCH 迭代 (高精度)
let sigma2 = math.bignumber(omega);
for (let t = 1; t < N; t++) {
  const eps2 = math.bignumber(returns[t] ** 2);
  sigma2 = math.add(
    math.bignumber(omega),
    math.add(
      math.multiply(math.bignumber(alpha), eps2),
      math.multiply(math.bignumber(beta), sigma2),
    ),
  );
}

// PCA 特征分解 (高精度 SVD)
const { U, S, V } = math.svd(math.matrix(covMatrix));
// S 对角矩阵 → 特征值 (64位精度)
