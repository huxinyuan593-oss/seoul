🔒 策略-执行隔离

量化引擎只算不碰：不持有私钥、不操作 UTXO、不写链。策略 BUG → UTXO 校验自动拦截。

🔄 原子性保证

数据库写入与链上广播通过 RequestID 关联。若链上广播失败，DB 记录标记 FAILED 并回滚 UTXO 锁。

⚡ 毫秒 vs 分钟

量化决策毫秒级完成；BTC 确认分钟级异步等待。UTXO 预锁 TTL=2h 弥合速度鸿沟。

🔍 确定性对账

RequestID 贯穿 信号→入库→Merkle→链上。全链路可追溯，任意节点可独立验证。

2. 8 大量化模型 — TypeScript API 接口

① GARCH(1,1) — 波动率预测

/**
 * GARCH(1,1) 波动率预测引擎
 * σ²ₜ = ω + α·ε²ₜ₋₁ + β·σ²ₜ₋₁
 *
 * 输出波动风险因子，供熔断控制器使用
 */
interface GARCHConfig {
  p: number;          // ARCH 阶数 (默认1)
  q: number;          // GARCH 阶数 (默认1)
  threshold: number;  // 熔断阈值 (默认0.05 = 年化5%)
}

interface GARCHResult {
  omega: number;          // ω — 长期方差
  alpha: number;          // α — 近期冲击反应系数
  beta: number;           // β — 波动持续性系数
  volatility: number;     // 当前年化波动率 (decimal)
  riskThreshold: number;  // 风险阈值
  isAbnormal: boolean;    // 是否异常 → 触发熔断
}

interface GARCHEngine {
  fit(returns: Float64Array): GARCHResult;
  predict(recentReturns: Float64Array): number;
  checkAbnormal(recentReturns: Float64Array): { isAbnormal: boolean; volatility: number };
}

② Kelly 公式 — 最优仓位管理

/**
 * Kelly Criterion 仓位计算
 * f* = (b·p - q) / b
 *
 * 支持 Full Kelly / Half Kelly / Quarter Kelly 三种保守度
 */
type KellyCriterion = 'FULL' | 'HALF' | 'QUARTER';

interface KellyConfig {
  criterion: KellyCriterion;  // 凯利分数类型
  maxLeverage: number;        // 最大杠杆上限 (默认1.0)
}

interface KellyResult {
  optimalFraction: number;    // f* 理论最优仓位比例
  adjustedFraction: number;   // 应用 Half/Quarter 后的实际仓位
  winProbability: number;     // p — 胜率
  odds: number;               // b — 赔率
  expectedValue: number;      // 期望值
  maxDrawdownRisk: number;    // 该仓位下的预估最大回撤
}

interface KellyPosition {
  size(winRate: number, odds: number, criterion: KellyCriterion): KellyResult;
}

③ HMM — 市场状态识别

/**
 * Hidden Markov Model — 市场状态推断
 * 隐状态: BULL (牛市) | BEAR (熊市) | RANGING (震荡)
 *
 * 根据状态自动切换策略：牛市中趋势跟随，熊市中保守，震荡中均值回归
 */
type MarketState = 'BULL' | 'BEAR' | 'RANGING';

interface HMMConfig {
  nStates: number;       // 隐状态数量 (默认3)
  covarianceType: 'full' | 'diag' | 'spherical';
  nIter: number;         // EM 算法迭代次数
}

interface HMMResult {
  currentState: MarketState;         // 当前推断的市场状态
  stateProbabilities: Float64Array;  // 各状态的后验概率
  transitionMatrix: Float64Array[];  // 状态转移矩阵 (nStates × nStates)
  stateMeans: Float64Array;          // 各状态的期望收益
  confidence: number;                // 推断置信度 0-1
}

interface HMMDetector {
  detect(features: Float64Array[]): HMMResult;
  /** 根据状态返回推荐策略 */
  recommendStrategy(state: MarketState): 'TREND_FOLLOWING' | 'MEAN_REVERSION' | 'CONSERVATIVE';
}

④ Z-Score — 统计套利信号

/**
 * Z-Score 均值回归套利
 * Zₜ = (εₜ - μ_ε) / σ_ε
 *
 * |Z| > entryThreshold → 开仓
 * |Z| < exitThreshold  → 平仓
 */
type ArbitrageSignal = 'LONG_SPREAD' | 'SHORT_SPREAD' | 'NEUTRAL';

interface ZScoreConfig {
  windowSize: number;        // 历史窗口大小
  entryThreshold: number;    // 入场阈值 (默认2.0)
  exitThreshold: number;     // 出场阈值 (默认0.5)
}

interface ZScoreResult {
  zScores: Float64Array;         // Z 值序列
  currentZ: number;              // 当前 Z 值
  signal: ArbitrageSignal;       // 当前信号
  spreadMean: number;            // μ_ε
  spreadStd: number;             // σ_ε
  confidence: number;            // 0-1
  halfLife: number;              // 均值回归半衰期 (周期数)
}

interface ZScoreArbitrage {
  compute(historical: Float64Array, current: Float64Array, config: ZScoreConfig): ZScoreResult;
}

4. 审计闭环 — generateMerkleAudit 核心函数

🔐 generateMerkleAudit — 全链路审计存证

/**
 * 默克尔审计存证引擎
 *
 * 将每日交易流水哈希生成 Merkle Root，写入 BTC 链上 OP_RETURN。
 * 任何人可独立验证数据库是否被篡改。
 *
 * 算法: 双 SHA-256 Merkle Tree (与 BTC 协议完全兼容)
 *
 * @param transactions 当日交易列表 (按 transactionId 排序)
 * @param btcClient    BTC RPC 客户端
 * @param fundingWIF   平台钱包私钥 (WIF格式)
 * @returns 锚定交易 ID + Merkle Root + 所有交易的证明路径
 */
interface Transaction {
  requestId: string;       // UUIDv7 — 来自量化引擎
  clientId: string;
  symbol: string;
  side: 'BUY' | 'SELL';
  price: number;
  size: number;            // BTC 数量
  btcTxid: string;         // 链上交易哈希
  timestamp: number;       // Unix 毫秒
  fee: number;             // 手续费
}

interface MerkleProof {
  txHash: string;          // 待验证的交易数据哈希
  siblings: string[];      // 路径上的兄弟哈希 (从叶子到根)
  index: number;           // 叶子位置
  root: string;            // Merkle Root
}

interface AuditRecord {
  batchDate: string;              // YYYY-MM-DD
  merkleRoot: string;             // 64 hex chars
  transactionCount: number;
  btcTxid: string;                // 锚定 BTC 交易 ID
  blockHeight: number;
  proofs: Map<string, MerkleProof>; // requestId → proof
  verified: boolean;
}

/**
 * 核心审计函数 — 请在生产环境使用此实现
 */
class MerkleAuditEngine {
  /**
   * 双 SHA-256 哈希 (BTC 标准)
   */
  static doubleSHA256(data: string): string {
    const { createHash } = require('crypto');
    const hash1 = createHash('sha256').update(data).digest();
    return createHash('sha256').update(hash1).digest('hex');
  }

  /**
   * 构建 Merkle Tree，返回各层
   * 非 2^n 叶子 → 复制最后一个补齐 (BTC 方式)
   */
  static buildLayers(txDataList: string[]): string[][] {
    // ① 对每笔交易数据做双 SHA-256
    let leaves = txDataList.map(d => this.doubleSHA256(d));

    // ② 补齐到 2^n
    const nextPow2 = 2 ** Math.ceil(Math.log2(Math.max(leaves.length, 2)));
    while (leaves.length < nextPow2) {
      leaves.push(leaves[leaves.length - 1]);
    }

    const layers: string[][] = [leaves];

    // ③ 逐层构建: H(left + right) 串联后双SHA256
    while (layers[layers.length - 1].length > 1) {
      const current = layers[layers.length - 1];
      const next: string[] = [];
      for (let i = 0; i < current.length; i += 2) {
        next.push(this.doubleSHA256(current[i] + current[i + 1]));
      }
      layers.push(next);
    }

    return layers;
  }

  /**
   * 生成 Merkle Proof — O(log₂N) 路径
   */
  static generateProof(layers: string[][], txIndex: number): MerkleProof {
    const siblings: string[] = [];
    let idx = txIndex;

    for (const layer of layers.slice(0, -1)) {
      const siblingIdx = idx % 2 === 0 ? idx + 1 : idx - 1;
      siblings.push(layer[siblingIdx] ?? layer[idx]);
      idx = Math.floor(idx / 2);
    }

    return {
      txHash: layers[0][txIndex],
      siblings,
      index: txIndex,
      root: layers[layers.length - 1][0],
    };
  }

  /**
   * 验证 Merkle Proof — O(log₂N) 重算
   */
  static verifyProof(proof: MerkleProof): boolean {
    let current = proof.txHash;
    let idx = proof.index;

    for (const sibling of proof.siblings) {
      current = idx % 2 === 0
        ? this.doubleSHA256(current + sibling)
        : this.doubleSHA256(sibling + current);
      idx = Math.floor(idx / 2);
    }

    return current === proof.root;
  }

  /**
   * 🔐 generateMerkleAudit — 全链路审计存证
   *
   * 1. 交易数据序列化 + 排序 (确定性)
   * 2. 构建 Merkle Tree → Root
   * 3. 为每笔交易生成 Merkle Proof
   * 4. OP_RETURN 锚定到 BTC 链上
   * 5. 写入 PostgreSQL audit_records 表
   */
  async generateMerkleAudit(
    transactions: Transaction[],
    btcClient: IBitcoinClient,
    fundingWIF: string,
  ): Promise<AuditRecord> {
    // ── Step 1: 确定性序列化 ──
    const sorted = [...transactions].sort((a, b) =>
      a.requestId.localeCompare(b.requestId)
    );

    const txDataList = sorted.map(tx =>
      `${tx.requestId}|${tx.btcTxid}|${tx.side}|` +
      `${tx.price}|${tx.size}|${tx.timestamp}|${tx.fee}`
    );

    // ── Step 2: 构建 Merkle Tree ──
    const layers = MerkleAuditEngine.buildLayers(txDataList);
    const merkleRoot = layers[layers.length - 1][0];

    // ── Step 3: 为每笔交易生成证明 ──
    const proofs = new Map<string, MerkleProof>();
    for (let i = 0; i < sorted.length; i++) {
      proofs.set(sorted[i].requestId,
        MerkleAuditEngine.generateProof(layers, i));
    }

    // ── Step 4: OP_RETURN 锚定 ──
    // 脚本: 0x6a 0x20 <32-byte Merkle Root>
    const rootBytes = Buffer.from(merkleRoot, 'hex');
    const opReturnScript = Buffer.concat([
      Buffer.from([0x6a, 0x20]),
      rootBytes,
    ]);

    // 构造锚定交易 (在生产环境使用 bitcoinlib)
    const rawTx = await this.buildAnchorTx(
      btcClient, fundingWIF, opReturnScript
    );

    // 广播
    const anchorTxid = await btcClient.sendRawTransaction(rawTx);

    // ── Step 5: 写入审计记录 ──
    const batchDate = new Date().toISOString().split('T')[0];
    const record: AuditRecord = {
      batchDate,
      merkleRoot,
      transactionCount: transactions.length,
      btcTxid: anchorTxid,
      blockHeight: 0, // 等待确认后更新
      proofs,
      verified: true,
    };

    // INSERT INTO audit_records ...
    // UPDATE transactions SET merkle_root = $1 WHERE ...

    return record;
  }

  /**
   * 🔍 防篡改验证: 定时重算 Root vs 链上 OP_RETURN
   *
   * 若 computed_root ≠ onchain_root → 🚨 数据库被篡改！
   */
  async verifyIntegrity(
    transactions: Transaction[],
    onchainAnchorTxid: string,
    btcClient: IBitcoinClient,
  ): Promise<{ valid: boolean; computedRoot: string; onchainRoot: string }> {
    // ① 重算 Merkle Root
    const sorted = [...transactions].sort((a, b) =>
      a.requestId.localeCompare(b.requestId)
    );
    const txDataList = sorted.map(tx =>
      `${tx.requestId}|${tx.btcTxid}|${tx.side}|` +
      `${tx.price}|${tx.size}|${tx.timestamp}|${tx.fee}`
    );
    const layers = MerkleAuditEngine.buildLayers(txDataList);
    const computedRoot = layers[layers.length - 1][0];

    // ② 从链上读取 OP_RETURN
    const rawTx = await btcClient.getRawTransaction(onchainAnchorTxid);
